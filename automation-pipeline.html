# Define the path to the SVN working directory
$workingDirectory = "C:\Users\akraj\Desktop\documentation"

$tortoiseProcPath = "C:\Program Files\TortoiseSVN\bin\TortoiseProc.exe"

# Check if the TortoiseProc executable exists
if (Test-Path $tortoiseProcPath) {
  # Run SVN update
  $process = Start-Process -FilePath $tortoiseProcPath `
  -ArgumentList "/command:update", "/path:$workingDirectory", "/closeonend:1" `
  -WindowStyle Minimized -PassThru

  $process | Wait-Process

  Write-Host "OK"
} else {
  Write-Output "TortoiseProc executable not found at $tortoiseProcPath"
}

sleep 10

# Deploy build using Posh-SSH
Import-Module Posh-SSH

Function Invoke-SCP {
  param(
    [string]$LocalPath,
    [string]$RemotePath,
    [string]$Server,
    [string]$Password,
    [string]$User,
    [int]$Port = 22
  )

  $pass = $Password | ConvertTo-SecureString -AsPlainText -Force
  $MySecureCreds = New-Object System.Management.Automation.PSCredential($User, $pass)

  $session = New-SFTPSession -ComputerName $Server -Credential $MySecureCreds -Port $Port

  if ($session) {
    Write-Host "Connected to $Server. Initiating file transfer..."

    $files = Get-ChildItem -Path $LocalPath
    $files | % {
      $fullname = $_.fullname
      $filename = $_.Name
      sleep 5
      Write-Host "Attempting to transfer file: $fullname to $remoteDirectory"
      $output = Set-SFTPItem -SessionId $session.SessionId `
      -Path "$fullname" -Destination "$remoteDirectory" -Force
      $output
      Write-Host "File transfer complete."
    }
    Remove-SFTPSession -SessionId $session.SessionId
  } else {
    Write-Host "Failed to connect to $Server."
  }
}
