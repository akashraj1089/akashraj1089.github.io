# Full script excerpt (trimmed for portfolio) Function Run-RoboHelp { param ($ProjectFilePath, $LogFilePath, $OutputPath) Start-Process -FilePath $roboHelpPath ` -ArgumentList "--cl `"$ProjectFilePath`" -o help --log `"$LogFilePath`" -p `"$OutputPath`"" ` -NoNewWindow -Wait } Function Invoke-SCP { param ($LocalPath, $RemotePath) Set-SFTPItem -SessionId $session.SessionId ` -Path $LocalPath ` -Destination $RemotePath -Force }
